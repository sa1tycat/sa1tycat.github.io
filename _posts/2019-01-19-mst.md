# 定理的推论

定理好像没什么用，主要是推论有用：

给定一张无向图 $G=(V,E)$，$n=|V|，m=|E|$ .从 $E$ 中选出 $k<n-1$ 条边构成 $G$ 的一个生成森林。若再从剩余的 $m-k$ 条边中选 $n-1-k$ 条边添加到生成森林中，使起成为 $G$ 的生成树，并且选出的边的权值之后最小，则该生成树一定包含这 $m-k$ 条边中连接生成森林的两个不连通结点的权值最小的边。

这就暗示了，最小生成树一定含有两个结点间权值最小的边。这也因此使得下面的两种算法奏效。

## Prim算法

```c++
#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#define MAXN 9010
#define INF 1e8
using namespace std;
int n,m;
int e[MAXN][MAXN],d[MAXN],visited[MAXN];
void Prim(){
    for(int i=1;i<=n;++i){
        d[i]=INF; visited[i]=0;
    }
    d[1]=0;
    for(int i=1;i<n;++i){
        int x=0;
        for(int j=1;j<=n;++j)
            if(!visited[j]&&(x==0||d[j]<d[x])) x=j;
        visited[x]=1;
        for(int y=1;y<=n;++y)
            if(!visited[y]) d[y]=min(d[y],e[x][y]);
    }
    return;
}

int main(){
    cin>>n>>m;
    for(int i=1;i<=n;++i)
        for(int j=1;j<=n;++j)
            e[i][j]=i==j?0:INF;
    int u,v,w;
    for(int i=1;i<=m;++i){
        scanf("%d%d%d",&u,&v,&w);
        e[u][v]=e[v][u]=w;
    }
    Prim();
    int ans=0;
    for(int i=2;i<=n;++i){
        ans+=d[i];
    }
    printf("%d",ans);
    return 0;
}
```

## Kruskal算法

```c++
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#define MAXN 101
#define MAXM 301
using namespace std;
int fa[MAXN];
int find(int x){
    if(x!=fa[x]) fa[x]=find(fa[x]);
    return fa[x];
}
struct node{
    int u,v,w;
}e[MAXM];
int n,m;
int d[MAXN],path[MAXN]; //path records the number it passes
bool cmp(const node &a,const node &b){ return a.w<b.w; }

int main(){
    cin>>n>>m;
    for(int i=1;i<=n;++i)
        fa[i]=i;
    for(int i=1;i<=m;++i)
        scanf("%d%d%d",&e[i].u,&e[i].v,&e[i].w);
    sort(e+1,e+m+1,cmp);
    for(int i=1;i<=n;++i) // at first each node is a set
        fa[i]=i;
    int k=0,ans=0;
    for(int i=1;i<=m;++i){
        if(find(e[i].u)!=find(e[i].v)){
            d[++k]=e[i].w;
            path[k]=i;
            ans+=e[i].w;
            fa[find(e[i].u)]=find(e[i].v);
        }
        else{
            continue;
        }
    }
    printf("%d\n",ans);
    for(int i=1;i<=k;++i){
        printf("%d %d\n",e[path[i]].u,e[path[i]].v);
    }
    return 0;
}
```
